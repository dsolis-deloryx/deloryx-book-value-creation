\chapter{Business Metrics Reference}\label{ch:metrics-catalog}

% Scope: dense metric-card catalog (formula + what-it-tells-you + benchmark + trap),
%   grouped by domain, with a flagged vanity-metrics section. Reference, not narrative.

This chapter is a lookup catalog, not an argument. Each card gives the formula, what the number actually tells you, a defensible benchmark where one exists, and the trap that makes the metric lie. Definitions follow \textcite{bendle2020marketing}; the derivations live in the home chapters --- CAC and LTV in \cref{ch:customer-economics}, ROAS/CPA mechanics in \cref{ch:paid-acquisition-platforms}, recurring-revenue metrics in \cref{ch:business-models}. A benchmark is an order-of-magnitude anchor, never a target in itself.

\section{Unit Economics}

\begin{multicols}{2}
\begin{metriccard}{CAC}{\dfrac{\text{Sales \& marketing spend}}{\text{New customers}}}
  \mcfield{Tells you}{Fully loaded cost to win one customer}
  \mcfield{Benchmark}{Judge only against LTV (target LTV:CAC $\ge 3$)}
  \mcfield{Trap}{Excluding salaries/tooling understates it badly}
\end{metriccard}

\begin{metriccard}{LTV}{\dfrac{m}{r-g}}
  \mcfield{Tells you}{Discounted lifetime contribution of a customer ($m$ = margin)}
  \mcfield{Benchmark}{$\ge 3\times$ CAC}
  \mcfield{Trap}{Using revenue not margin; same algebra as \cref{eq:gordon}}
\end{metriccard}

\begin{metriccard}{LTV:CAC}{\dfrac{\text{LTV}}{\text{CAC}}}
  \mcfield{Tells you}{Capital efficiency of acquisition}
  \mcfield{Benchmark}{3--5 healthy; $<1$ unprofitable}
  \mcfield{Trap}{$>5$ usually means under-investing in growth}
\end{metriccard}

\begin{metriccard}{CAC Payback}{\dfrac{\text{CAC}}{\text{monthly margin/customer}}}
  \mcfield{Tells you}{Months to recoup acquisition cost}
  \mcfield{Benchmark}{$<12$ mo (SMB); $<24$ mo (enterprise)}
  \mcfield{Trap}{Ignores customers who churn before payback}
\end{metriccard}

\begin{metriccard}{Contribution Margin}{\text{Price}-\text{variable cost}}
  \mcfield{Tells you}{Cash each unit throws off after variable cost}
  \mcfield{Benchmark}{Positive and rising with scale}
  \mcfield{Trap}{Confusing it with gross margin}
\end{metriccard}

\begin{metriccard}{Gross Margin}{\dfrac{\text{Revenue}-\text{COGS}}{\text{Revenue}}}
  \mcfield{Tells you}{Structural profitability of the offering}
  \mcfield{Benchmark}{\qty{70}{\percent}+ software; lower for physical}
  \mcfield{Trap}{Burying hosting/support in opex to flatter it}
\end{metriccard}
\end{multicols}

\section{SaaS \& Recurring Revenue}

\begin{multicols}{2}
\begin{metriccard}{MRR}{\textstyle\sum \text{monthly subscription value}}
  \mcfield{Tells you}{Predictable monthly run-rate}
  \mcfield{Benchmark}{Growth rate matters more than level}
  \mcfield{Trap}{Counting one-off or usage fees as recurring}
\end{metriccard}

\begin{metriccard}{ARR}{\text{MRR}\times 12}
  \mcfield{Tells you}{Annualised recurring run-rate}
  \mcfield{Trap}{Annualising non-recurring revenue}
\end{metriccard}

\begin{metriccard}{Logo Churn}{\dfrac{\text{customers lost}}{\text{customers at start}}}
  \mcfield{Tells you}{Customer (count) attrition rate}
  \mcfield{Benchmark}{$<\qty{2}{\percent}$/mo SMB}
  \mcfield{Trap}{Ignores that lost logos may be tiny accounts}
\end{metriccard}

\begin{metriccard}{NRR}{\dfrac{\text{start}+\text{expansion}-\text{contraction}-\text{churn}}{\text{start MRR}}}
  \mcfield{Tells you}{Growth from the existing base alone}
  \mcfield{Benchmark}{$>\qty{100}{\percent}$ good; $>\qty{120}{\percent}$ elite}
  \mcfield{Trap}{A high NRR can mask heavy logo churn}
\end{metriccard}

\begin{metriccard}{Rule of 40}{\text{growth\,\%}+\text{profit\,\%}}
  \mcfield{Tells you}{Balance of growth and profitability}
  \mcfield{Benchmark}{$\ge 40$}
  \mcfield{Trap}{Gaming it with one-time margin}
\end{metriccard}

\begin{metriccard}{Burn Multiple}{\dfrac{\text{net burn}}{\text{net new ARR}}}
  \mcfield{Tells you}{Cash burned per \$1 of new ARR}
  \mcfield{Benchmark}{$<1.5$ good; $<1$ excellent}
  \mcfield{Trap}{Reads worse in deliberate land-grab phases}
\end{metriccard}
\end{multicols}

\section{Funnel \& Marketing}

\begin{multicols}{2}
\begin{metriccard}{CTR}{\dfrac{\text{Clicks}}{\text{Impressions}}}
  \mcfield{Tells you}{Creative + targeting resonance}
  \mcfield{Trap}{High CTR with low CVR = wrong audience}
\end{metriccard}

\begin{metriccard}{CPC}{\dfrac{\text{Cost}}{\text{Clicks}}}
  \mcfield{Tells you}{Price of a visit}
  \mcfield{Trap}{Lowered by Quality Score (\cref{eq:ecpc}); don't optimise it over CPA}
\end{metriccard}

\begin{metriccard}{CPM}{\dfrac{\text{Cost}}{\text{Impressions}}\times 1000}
  \mcfield{Tells you}{Cost to reach a thousand impressions}
  \mcfield{Trap}{Cheap CPM on a worthless audience}
\end{metriccard}

\begin{metriccard}{CPA / CPL}{\dfrac{\text{Ad spend}}{\text{Conversions}}}
  \mcfield{Tells you}{Cost per acquired action or lead}
  \mcfield{Benchmark}{Below the LTV-implied ceiling (\cref{ch:paid-acquisition-platforms})}
  \mcfield{Trap}{Counting low-intent actions as conversions}
\end{metriccard}

\begin{metriccard}{ROAS}{\dfrac{\text{Conversion value}}{\text{Ad spend}}}
  \mcfield{Tells you}{Revenue returned per ad dollar}
  \mcfield{Benchmark}{3--8$\times$ e-commerce}
  \mcfield{Trap}{Platform-reported ROAS over-credits (\cref{ch:attribution-and-measurement})}
\end{metriccard}

\begin{metriccard}{MER}{\dfrac{\text{Total revenue}}{\text{Total marketing spend}}}
  \mcfield{Tells you}{Blended, whole-business ad efficiency}
  \mcfield{Trap}{Hides which channel actually drove revenue}
\end{metriccard}

\begin{metriccard}{AOV}{\dfrac{\text{Revenue}}{\text{Orders}}}
  \mcfield{Tells you}{Average basket size}
  \mcfield{Trap}{The mean hides a long, skewed tail}
\end{metriccard}

\begin{metriccard}{Conversion Rate}{\dfrac{\text{Conversions}}{\text{Sessions}}}
  \mcfield{Tells you}{Funnel efficiency at a given step}
  \mcfield{Trap}{Site-wide CVR is meaningless without a segment}
\end{metriccard}
\end{multicols}

\section{Ad-Platform}

\begin{multicols}{2}
\begin{metriccard}{eCPM}{\dfrac{\text{Total cost}}{\text{Impressions}}\times 1000}
  \mcfield{Tells you}{All-in realised cost per thousand, across buy types}
  \mcfield{Trap}{Comparing eCPM across very different placements}
\end{metriccard}

\begin{metriccard}{Impression Share}{\dfrac{\text{impressions won}}{\text{impressions eligible}}}
  \mcfield{Tells you}{Auction coverage and remaining headroom}
  \mcfield{Benchmark}{Split lost-IS into budget vs rank to find the bottleneck}
  \mcfield{Trap}{Chasing \qty{100}{\percent} share at any price}
\end{metriccard}

\begin{metriccard}{ACOS}{\dfrac{\text{Ad spend}}{\text{Sales}}}
  \mcfield{Tells you}{Ad cost as a share of sales (inverse ROAS)}
  \mcfield{Trap}{Target ACOS must be set from the product margin}
\end{metriccard}

\begin{metriccard}{iROAS}{\dfrac{\text{incremental revenue}}{\text{ad spend}}}
  \mcfield{Tells you}{Causal return, not correlation}
  \mcfield{Trap}{Requires a holdout/geo test (\cref{ch:attribution-and-measurement})}
\end{metriccard}
\end{multicols}

\section{Product \& Engagement}

\begin{multicols}{2}
\begin{metriccard}{Stickiness}{\dfrac{\text{DAU}}{\text{MAU}}}
  \mcfield{Tells you}{How habitual the product is}
  \mcfield{Benchmark}{$>\qty{20}{\percent}$ decent; $>\qty{50}{\percent}$ elite}
  \mcfield{Trap}{Bot/duplicate accounts inflate it}
\end{metriccard}

\begin{metriccard}{D30 Retention}{\dfrac{\text{active at day 30}}{\text{cohort size}}}
  \mcfield{Tells you}{Durable value of an acquisition cohort}
  \mcfield{Trap}{Averaging across acquisition sources (\cref{ch:digital-funnel-and-crm})}
\end{metriccard}

\begin{metriccard}{NPS}{\%\,\text{promoters}-\%\,\text{detractors}}
  \mcfield{Tells you}{Coarse word-of-mouth proxy}
  \mcfield{Trap}{Reading an 11-point survey as a precise number}
\end{metriccard}

\begin{metriccard}{Activation Rate}{\dfrac{\text{reached aha-moment}}{\text{signups}}}
  \mcfield{Tells you}{Onboarding effectiveness}
  \mcfield{Trap}{Defining ``activated'' too loosely to flatter it}
\end{metriccard}
\end{multicols}

\section{Vanity Metrics --- What They Hide}

These move with effort, not outcomes. Each is flagged because it actively misleads, not merely imprecise; pair it with the outcome metric noted.

\begin{multicols}{2}
\begin{metriccard}[colframe=red!65!black, colbacktitle=red!65!black]{Raw Impressions}{\text{total ad views}}
  \mcfield{Tells you}{Almost nothing on its own}
  \mcfield{Vanity}{Scales with spend; pair with reach \& frequency}
\end{metriccard}

\begin{metriccard}[colframe=red!65!black, colbacktitle=red!65!black]{Email Open Rate}{\dfrac{\text{opens}}{\text{delivered}}}
  \mcfield{Tells you}{Little since Apple MPP}
  \mcfield{Vanity}{Privacy proxies pre-fetch the pixel; track clicks/replies}
\end{metriccard}

\begin{metriccard}[colframe=red!65!black, colbacktitle=red!65!black]{Last-Click ROAS}{\dfrac{\text{last-click rev.}}{\text{spend}}}
  \mcfield{Tells you}{Who closed, not who caused}
  \mcfield{Vanity}{Ignores assist + view-through; use iROAS (\cref{ch:attribution-and-measurement})}
\end{metriccard}

\begin{metriccard}[colframe=red!65!black, colbacktitle=red!65!black]{Unqualified Leads}{\text{form fills}}
  \mcfield{Tells you}{Interest, not demand}
  \mcfield{Vanity}{A lead is not a customer; feed \emph{qualified} value back (\cref{ch:paid-acquisition-platforms})}
\end{metriccard}

\begin{metriccard}[colframe=red!65!black, colbacktitle=red!65!black]{Follower Count}{\text{social followers}}
  \mcfield{Tells you}{Audience size, not value}
  \mcfield{Vanity}{No revenue link; track the engaged, converting cohort}
\end{metriccard}

\begin{metriccard}[colframe=red!65!black, colbacktitle=red!65!black]{App Downloads}{\text{installs}}
  \mcfield{Tells you}{Trial, not adoption}
  \mcfield{Vanity}{Meaningless without activation + retention}
\end{metriccard}
\end{multicols}
